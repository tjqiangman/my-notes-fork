
2. 面向对象 -- 私有变量无法在类外部被访问
    ```python
    class Test(object):
        
        def __init__(self):
            self.__name = ''
            self.__age = 0
            self.__weight = 60.0
            
        def set_name(self, name: str):
            self.__name = name
            
        def set_age(self, age: int):
            self.__age = age
        
        def get_name(self,):
            return self.__name
        
        def get_age(self,):
            return self.__age
            
        def get_weight(self,):
            return self.__weight
            
            
    def main():
        print('Hello, Python! This is a script which tests.')
        t = Test()
        t.set_age(20)
        t.set_name('JinYu')
        
        print(t.__age)  # 报错
        
        print(f'My name is {t.get_name()}, I am {t.get_age()} years old, and I have a {t.get_weight()} weight body. Yeah!')
        return 0
    
    
    if __name__ == '__main__':
        main()
    
    ```
